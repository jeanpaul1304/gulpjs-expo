var math, cubes;
// Objetos
math = {
  root: Math.sqrt,
  square: square,
  cube: function(x){
    return x * square(x)
  }
};

// Existencia
if (typeof elvis !== "undefined" && elvis !== null){
  alert('Epa! Coffescript');
}

// Array comprehensions
cubes = (function() {
  var i, len, results;
  results = [];
  for (i = 0, len = list.length; i < len; i++) {
    num = list[i];
    results.push(math.cube(num));
  }
  return results;
})();
